````markdown
### 3️⃣ إصلاح MagneticButton — استخدم Link (حرج — 10 دقائق)

- [ ] افتح `src/components/ui-premium/magnetic-button.tsx`
- [ ] **استبدل الكود الكامل بـ:**
  ```tsx
  'use client'

  import { useRef, type ReactNode, type MouseEvent } from 'react'
  import { motion, useMotionValue, useSpring } from 'framer-motion'
  import { Link } from '@/i18n/routing'

  interface MagneticButtonProps {
    children: ReactNode
    className?: string
    strength?: number
    onClick?: () => void
    href?: string
  }

  export function MagneticButton({
    children,
    className = '',
    strength = 0.3,
    onClick,
    href,
  }: MagneticButtonProps) {
    const ref = useRef<HTMLDivElement>(null)
    const x = useMotionValue(0)
    const y = useMotionValue(0)

    const springX = useSpring(x, { stiffness: 200, damping: 20 })
    const springY = useSpring(y, { stiffness: 200, damping: 20 })

    const handleMouseMove = (e: MouseEvent<HTMLDivElement>) => {
      if (!ref.current) return
      const rect = ref.current.getBoundingClientRect()
      const centerX = rect.left + rect.width / 2
      const centerY = rect.top + rect.height / 2
      const distX = e.clientX - centerX
      const distY = e.clientY - centerY
      x.set(distX * strength)
      y.set(distY * strength)
    }

    const handleMouseLeave = () => {
      x.set(0)
      y.set(0)
    }

    const content = (
      <motion.div
        ref={ref}
        onMouseMove={handleMouseMove}
        onMouseLeave={handleMouseLeave}
        style={{ x: springX, y: springY }}
        className={className}
      >
        {children}
      </motion.div>
    )

    if (href) {
      return (
        <Link href={href} onClick={onClick} className="inline-block">
          {content}
        </Link>
      )
    }

    return (
      <button onClick={onClick} className="inline-block bg-transparent border-0 p-0 cursor-pointer">
        {content}
      </button>
    )
  }
  ```

  'use client'

  import { useState, type ReactNode } from 'react'
  import { useTranslations } from 'next-intl'
  import { Button } from '@/components/ui/button'
  import { AlertTriangle, X } from 'lucide-react'

  interface ConfirmDeleteProps {
    trigger: ReactNode
    itemName: string
    onConfirm: () => Promise<void> | void
  }

  export function ConfirmDelete({ trigger, itemName, onConfirm }: ConfirmDeleteProps) {
    const t = useTranslations('admin.confirmDelete')
    const [open, setOpen] = useState(false)
    const [deleting, setDeleting] = useState(false)

    const handleConfirm = async () => {
      setDeleting(true)
      try {
        await onConfirm()
        setOpen(false)
      } finally {
        setDeleting(false)
      }
    }

    return (
      <>
        <div onClick={() => setOpen(true)} className="cursor-pointer">
          {trigger}
        </div>

        {open && (
          <div
            className="fixed inset-0 z-[100] flex items-center justify-center bg-black/50 p-4"
            onClick={() => !deleting && setOpen(false)}
          >
            <div
              className="bg-card rounded-xl shadow-xl max-w-sm w-full p-6"
              onClick={(e) => e.stopPropagation()}
            >
              <div className="flex items-start justify-between mb-4">
                <div className="w-12 h-12 rounded-full bg-red-100 flex items-center justify-center">
                  <AlertTriangle className="w-6 h-6 text-red-600" />
                </div>
                <button
                  onClick={() => !deleting && setOpen(false)}
                  className="text-muted-foreground hover:text-foreground"
                  aria-label={t('cancel')}
                >
                  <X className="w-5 h-5" />
                </button>
              </div>

              <h3 className="text-lg font-bold text-foreground mb-2">
                {t('title')}
              </h3>
              <p className="text-sm text-muted-foreground mb-6">
                {t('message', { name: itemName })}
              </p>

              <div className="flex gap-3">
                <Button
                  variant="outline"
                  className="flex-1 border-border"
                  onClick={() => setOpen(false)}
                  disabled={deleting}
                >
                  {t('cancel')}
                </Button>
                <Button
                  className="flex-1 bg-red-600 hover:bg-red-700 text-white"
                  onClick={handleConfirm}
                  disabled={deleting}
                >
                  {deleting ? t('processing') : t('confirm')}
                </Button>
              </div>
            </div>
          </div>
        )}
      </>
    )
  }
  ```

### 6️⃣ إصلاح Toast Provider — a11y + Animation (عالي — ساعة)

- [ ] افتح `src/components/providers/toast-provider.tsx`
- [ ] **استبدل الكود الكامل بـ:**
  ```tsx
  'use client'

  import { createContext, useContext, useState, useCallback, type ReactNode } from 'react'
  import { CheckCircle2, XCircle, AlertCircle, X } from 'lucide-react'
  import { AnimatePresence, motion } from 'framer-motion'

  type ToastType = 'success' | 'error' | 'warning'

  interface Toast {
    id: string
    type: ToastType
    message: string
  }

  interface ToastContextValue {
    showToast: (type: ToastType, message: string) => void
  }

  const ToastContext = createContext<ToastContextValue | null>(null)

  export function ToastProvider({ children }: { children: ReactNode }) {
    const [toasts, setToasts] = useState<Toast[]>([])

    const removeToast = useCallback((id: string) => {
      setToasts((prev) => prev.filter((t) => t.id !== id))
    }, [])

    const showToast = useCallback((type: ToastType, message: string) => {
      const id = Math.random().toString(36).slice(2, 11)
      setToasts((prev) => [...prev, { id, type, message }])

      // Errors stay longer (8s), others 4s
      const duration = type === 'error' ? 8000 : 4000
      setTimeout(() => {
        removeToast(id)
      }, duration)
    }, [removeToast])

    return (
      <ToastContext.Provider value={{ showToast }}>
        {children}
        <div
          className="fixed bottom-4 end-4 z-[100] space-y-2"
          style={{
            paddingBottom: 'env(safe-area-inset-bottom, 1rem)',
            paddingRight: 'env(safe-area-inset-right, 1rem)',
          }}
          role="region"
          aria-label="Notifications"
        >
          <AnimatePresence>
            {toasts.map((toast) => (
              <motion.div
                key={toast.id}
                initial={{ opacity: 0, y: 20, scale: 0.9 }}
                animate={{ opacity: 1, y: 0, scale: 1 }}
                exit={{ opacity: 0, y: 20, scale: 0.9 }}
                transition={{ duration: 0.2 }}
                className={`flex items-center gap-3 px-4 py-3 rounded-lg shadow-lg text-white max-w-sm ${
                  toast.type === 'success'
                    ? 'bg-green-600'
                    : toast.type === 'error'
                    ? 'bg-red-600'
                    : 'bg-yellow-600'
                }`}
                role={toast.type === 'error' ? 'alert' : 'status'}
                aria-live={toast.type === 'error' ? 'assertive' : 'polite'}
              >
                {toast.type === 'success' && <CheckCircle2 className="w-5 h-5 shrink-0" />}
                {toast.type === 'error' && <XCircle className="w-5 h-5 shrink-0" />}
                {toast.type === 'warning' && <AlertCircle className="w-5 h-5 shrink-0" />}
                <span className="text-sm flex-1">{toast.message}</span>
                <button
                  onClick={() => removeToast(toast.id)}
                  aria-label="Close notification"
                  className="text-white/80 hover:text-white shrink-0"
                >
                  <X className="w-4 h-4" />
                </button>
              </motion.div>
            ))}
          </AnimatePresence>
        </div>
      </ToastContext.Provider>
    )
  }

  export function useToast() {
    const ctx = useContext(ToastContext)
    if (!ctx) throw new Error('useToast must be used within ToastProvider')
    return ctx
  }
  ```

```markdown
# Last Unique Touch — Design Rules (Do Not Violate)

> هذه القواعد مستخلصة من تقرير Tri-AI v3.0. أي مخالفة لها = bug مؤكد.

## 1. i18n — لا نصوص hardcoded
- ❌ لا تستخدم `'KWD/day'`, `'KWD'`, `'Subject'`, أو أي نص آخر hardcoded في JSX
- ✅ كل النصوص تمر عبر `t('key')` من `next-intl`
- ✅ كل مفتاح في `ar.json` يجب أن يكون في `en.json` أيضاً
- ✅ Admin panel也不例外 — كل النصوص مترجمة

## 2. الألوان — مصدر واحد للحقيقة
- ❌ لا تخترع hex codes جديدة في JSX
- ✅ استخدم CSS variables من `globals.css` (`--c-ink`, `--c-paper`, `--c-brand`, إلخ)
- ❌ لا تكرر نفس اللون في 3 أنظمة متغيرات
- ✅ استخدم متغيراً مصدراً واحداً (`--c-brand`) واجعل البقية يشيرون إليه

## 3. لغة التصميم — توحيد صارم
- ❌ لا تستخدم `rounded-xl` في صفحة و `borderRadius: '2px'` في صفحة أخرى لنفس العنصر
- ✅ اختر لغة واحدة لكل البطاقات/الكروت وطبّقها في كل مكان
- ❌ لا تستخدم `font-display` لعنوان في صفحة و `font-semibold` لنفس العنصر في صفحة أخرى
- ✅ عناوين المنتجات موحّدة عبر كل الصفحات

## 4. RTL — احترام الاتجاه
- ❌ لا تستخدم `left`/`right` في CSS أو `x: '100%'` ثابت في animations
- ✅ استخدم logical properties: `ms-`, `me-`, `ps-`, `pe-`, `start-`, `end-`
- ✅ animations تعتمد على `locale`: `const drawerX = locale === 'ar' ? '-100%' : '100%'`
- ✅ استخدم `ArrowLeft`/`ArrowRight` من lucide-react مع تبديل حسب الـ locale

## 5. التصفّح (Navigation) — استخدم Next.js Link
- ❌ لا تستخدم `<a href>` للروابط الداخلية
- ✅ استخدم `Link` من `@/i18n/routing` (يحافظ على locale + client-side navigation)
- ✅ MagneticButton وكل wrapper components يجب أن تستخدم `Link` داخلياً

## 6. الصور — استخدم next/image دائماً
- ❌ لا تستخدم `<img>` أبداً (لأسباب أداء + SEO)
- ✅ استخدم `next/image` مع `width`, `height`, `alt`, و `sizes`
- ✅ alt text وصفي للصور المعلوماتية، `alt=""` للزخرفية

```tsx
'use client'

import { useRef, useEffect, useMemo } from 'react'
import { useGLTF } from '@react-three/drei'
import { useFrame, useThree } from '@react-three/fiber'
import * as THREE from 'three'
import { useSpring, a } from '@react-spring/three'
import { useScrollStore } from '@/stores/scroll-store'

interface Model3DProps {
  url: string
  /** Position relative to brand card center [x, y, z] */
  position: [number, number, number]
  /** Initial rotation [x, y, z] in radians */
  rotation?: [number, number, number]
  /** Initial scale (1 = normal, larger = bigger) */
  scale?: number
  /** Phase offset for staggered animation (0, 1, 2...) */
  phaseOffset?: number
  /** Whether the model has loaded (for pop-in animation) */
  visible: boolean
  /** When the brand cards animation finishes (ms timestamp) */
  cardsAnimationCompleteAt: number
}

export function Model3D({
  url,
  position,
  rotation = [0, 0, 0],
  scale = 1,
  phaseOffset = 0,
  visible,
  cardsAnimationCompleteAt,
}: Model3DProps) {
  const groupRef = useRef<THREE.Group>(null)
  const { scene } = useGLTF(url, true, true)  // useDraco=true, useMeshOpt=true
  const { gl } = useThree()

  // Clone scene to avoid mutating cached version
  const clonedScene = useMemo(() => scene.clone(true), [scene])

  // Make all materials transparent (for fade out)
  useEffect(() => {
    clonedScene.traverse((child) => {
      if (child instanceof THREE.Mesh) {
        const mat = child.material as THREE.Material
        mat.transparent = true
        ;(mat as THREE.MeshStandardMaterial).opacity = 1
        child.castShadow = true
        child.receiveShadow = true
      }
    })
  }, [clonedScene])

  // Pop-in spring animation (elastic bounce)
  // ⚠️ Starts ONLY after cards animation completes + phaseOffset delay
  const springDelay = Math.max(0, cardsAnimationCompleteAt - Date.now()) + phaseOffset * 150

  const springs = useSpring({
    scale: visible ? scale : 0,
    opacity: visible ? 1 : 0,
    config: {
      tension: 280,
      friction: 18,
      mass: 1.2,
      clamp: false,
    },
    delay: springDelay,
  })

  // Scroll-driven animation (manual damp in useFrame)
  useFrame((_, delta) => {
    if (!groupRef.current) return

    const offset = useScrollStore.getState().offset
    // Hero section is roughly 0 to 0.12 of total page scroll
    const heroScrollProgress = Math.min(1, Math.max(0, (offset - phaseOffset * 0.03) / 0.12))

    // Phase 1 (0-0.4): scale down from 1 to 0.6
    const scalePhase = Math.min(1, heroScrollProgress / 0.4)
    const targetScale = THREE.MathUtils.lerp(scale, scale * 0.5, scalePhase)

    // Phase 2 (0.3-0.7): opacity fade from 1 to 0.5
    const opacityPhase = Math.min(1, Math.max(0, (heroScrollProgress - 0.3) / 0.4))
    const targetOpacity = THREE.MathUtils.lerp(1, 0.5, opacityPhase)

    // Phase 3 (0.6-1.0): lift up + rotate
    const liftPhase = Math.min(1, Math.max(0, (heroScrollProgress - 0.6) / 0.4))
    const targetY = THREE.MathUtils.lerp(position[1], position[1] + 1.5, liftPhase)
    const targetRotY = THREE.MathUtils.lerp(rotation[1], rotation[1] + Math.PI / 6, liftPhase)

    // Apply with damp for smooth, framerate-independent motion
    groupRef.current.scale.x = THREE.MathUtils.damp(
      groupRef.current.scale.x,
      targetScale * springs.scale.get(),
      8,
      delta
    )
    groupRef.current.scale.y = groupRef.current.scale.x
    groupRef.current.scale.z = groupRef.current.scale.x

    groupRef.current.position.y = THREE.MathUtils.damp(
      groupRef.current.position.y,
      targetY,
      8,
      delta
    )

    groupRef.current.rotation.y = THREE.MathUtils.damp(
      groupRef.current.rotation.y,
      targetRotY,
      6,
      delta
    )

    // Update material opacity (traverse once per frame)
    const finalOpacity = targetOpacity * springs.opacity.get()
    clonedScene.traverse((child) => {
      if (child instanceof THREE.Mesh) {
        ;(child.material as THREE.MeshStandardMaterial).opacity = finalOpacity
      }
    })
  })

  // WebGL context loss handler (critical for iOS 17+)
  useEffect(() => {
    const canvas = gl.domElement
    const handleContextLost = (e: Event) => {
      e.preventDefault()
      console.warn('[Model3D] WebGL context lost — will restore on focus')
    }
    const handleContextRestored = () => {
      console.log('[Model3D] WebGL context restored')
    }
    canvas.addEventListener('webglcontextlost', handleContextLost)
    canvas.addEventListener('webglcontextrestored', handleContextRestored)
    return () => {
      canvas.removeEventListener('webglcontextlost', handleContextLost)
      canvas.removeEventListener('webglcontextrestored', handleContextRestored)
    }
  }, [gl])

  return (
    <a.group
      ref={groupRef}
      position={position}
      rotation={rotation}
      scale={springs.scale}
    >
      <primitive object={clonedScene} />
    </a.group>
  )
}

// Preload all 3 models at module load (starts download immediately)
useGLTF.preload('/models/The_chair_and_the_table_compressed.glb')
useGLTF.preload('/models/The_unfinished_sofa_compressed.glb')
useGLTF.preload('/models/the_dance_floor_and_the_light_holder_compressed.glb')

'use client'

import { useEffect, useRef } from 'react'
import Lenis from 'lenis'
import { shouldEnable3D } from '@/lib/device-capabilities'

export function LenisProvider({ children }: { children: React.ReactNode }) {
  const lenisRef = useRef<Lenis | null>(null)

  useEffect(() => {
    // ✅ Enable on both PC + Mobile (was mobile-only before)
    if (!shouldEnable3D()) return

    const lenis = new Lenis({
      duration: 1.2,
      easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
      smoothWheel: true,
      // ⚠️ Critical: do NOT enable smoothTouch on iOS (causes jitter)
      smoothTouch: false,
      touchMultiplier: 1.5,
      infinite: false,
    })

    lenisRef.current = lenis

    let rafId: number
    const raf = (time: number) => {
      lenis.raf(time)
      rafId = requestAnimationFrame(raf)
    }
    rafId = requestAnimationFrame(raf)

    return () => {
      cancelAnimationFrame(rafId)
      lenis.destroy()
      lenisRef.current = null
    }
  }, [])

  return <>{children}</>
}
````
