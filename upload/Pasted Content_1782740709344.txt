````markdown
## 🎯 الهدف

بناء **تجربة 3D scroll-driven** على الموبايل تتميز بـ:
1. **3 نماذج GLB** معروضة فوق بطاقات البراند الثلاث
2. **Pop-in animation** عند تحميل الصفحة (Spring physics مع elastic bounce)
3. **Scroll-down**: النموذج يصغر + يتلاشى + يرتفع/يدور (بالتدريج، ليس متزامناً)
4. **Scroll-up**: عكس الحركة برشاقة
5. **60fps مستقرة** على iPhone SE (2020) و الأجهزة المتوسطة
6. **احترام a11y**: prefers-reduced-motion + prefers-reduced-data
7. **معالجة WebGL context loss** على iOS Safari
8. **عدم حجب أحداث اللمس** عن البطاقات HTML أدناه

## 🏗️ القرار المعماري الحاسم

### ❌ لا تستخدم `<ScrollControls>` من Drei
السبب: `<ScrollControls>` ينشئ **حاوية HTML scroll أمام الـ canvas** تأخذ كل أحداث اللمس. هذا يمنع النقر على البطاقات HTML أدناه، ويتعارض مع متطلب `pointer-events-none`.

### ✅ استخدم Native Scroll + Zustand Store
- الصفحة تنزلق طبيعياً (native body scroll، يحتاجه SEO + a11y)
- Canvas ثابت بـ `position: fixed; pointer-events: none`
- نقرأ `window.scrollY` ونخزّنه في Zustand store (مرة واحدة لكل frame)
- في `useFrame`، نطبّق `THREE.MathUtils.damp` على المجسمات

هذا هو النمط الذي تستخدمه الاستوديوهات الكبيرة (unseen, immerssive, 14islands).

```typescript
import { create } from 'zustand';

interface ScrollState {
  /** Scroll offset normalized 0-1 (0 = top, 1 = bottom of page) */
  offset: number;
  /** Whether the hero section is in viewport */
  isHeroVisible: boolean;
  /** Set scroll offset (called once per frame from rAF) */
  setOffset: (offset: number) => void;
  /** Set hero visibility (called from IntersectionObserver) */
  setHeroVisible: (visible: boolean) => void;
}

export const useScrollStore = create<ScrollState>((set) => ({
  offset: 0,
  isHeroVisible: true,
  setOffset: (offset) => set({ offset }),
  setHeroVisible: (isHeroVisible) => set({ isHeroVisible }),
}));

/**
 * Initialize scroll tracking.
 * Call this ONCE in a top-level client component.
 * Uses requestAnimationFrame to throttle updates.
 */
export function initScrollTracking() {
  if (typeof window === 'undefined') return () => {};

  let rafId: number;
  let targetOffset = 0;

  const updateOffset = () => {
    const scrollY = window.scrollY;
    const maxScroll = document.documentElement.scrollHeight - window.innerHeight;
    targetOffset = maxScroll > 0 ? Math.min(1, Math.max(0, scrollY / maxScroll)) : 0;
    rafId = requestAnimationFrame(updateOffset);
  };

  const syncStore = () => {
    useScrollStore.getState().setOffset(targetOffset);
    rafId = requestAnimationFrame(syncStore);
  };

  // Start both rAF loops
  rafId = requestAnimationFrame(updateOffset);
  const syncId = requestAnimationFrame(syncStore);

  return () => {
    cancelAnimationFrame(rafId);
    cancelAnimationFrame(syncId);
  };
}
```

> ⚠️ **مهم:** هذا النمط يفصل بين قراءة scroll (سريعة) وتحديث Zustand (مرة واحدة لكل frame). يمنع الـ "scroll lag".

/**
 * Detect device capabilities to decide whether to enable 3D.
 * Returns false for low-end devices, reduced motion, or reduced data.
 */
export function shouldEnable3D(): boolean {
  if (typeof window === 'undefined') return false;

  // 1. Respect prefers-reduced-motion
  const prefersReducedMotion = window.matchMedia('(prefers-reduced-motion: reduce)').matches;
  if (prefersReducedMotion) return false;

  // 2. Respect prefers-reduced-data
  const prefersReducedData = window.matchMedia('(prefers-reduced-data: reduce)').matches;
  if (prefersReducedData) return false;

  // 3. Check WebGL support
  try {
    const canvas = document.createElement('canvas');
    const gl = canvas.getContext('webgl2') || canvas.getContext('webgl');
    if (!gl) return false;
  } catch {
    return false;
  }

  // 4. Device memory + cores (low-end detection)
  const memory = (navigator as any).deviceMemory || 4;
  const cores = navigator.hardwareConcurrency || 4;
  const isMobile = /Android|iPhone|iPad|iPod/i.test(navigator.userAgent);

  // Disable on very low-end mobiles (< 4GB RAM or < 4 cores)
  if (isMobile && (memory < 4 || cores < 4)) return false;

  return true;
}

/**
 * Detect if device is mobile (touch-based).
 */
export function isMobileDevice(): boolean {
  if (typeof window === 'undefined') return false;
  return window.matchMedia('(pointer: coarse)').matches;
}
```

### 4️⃣ إنشاء مكون النموذج 3D (Model3D)

- [ ] أنشئ `src/components/hero-3d/model-3d.tsx`:

```tsx
'use client';

import { useRef, useEffect, useMemo } from 'react';
import { useGLTF } from '@react-three/drei';
import { useFrame, useThree } from '@react-three/fiber';
import * as THREE from 'three';
import { useSpring, a } from '@react-spring/three';
import { useScrollStore } from '@/stores/scroll-store';

interface Model3DProps {
  url: string;
  /** Initial position [x, y, z] */
  position: [number, number, number];
  /** Initial rotation [x, y, z] in radians */
  rotation?: [number, number, number];
  /** Initial scale */
  scale?: number;
  /** Phase offset for staggered animation (0-1) */
  phaseOffset?: number;
  /** Whether the model has loaded (for pop-in animation) */
  visible: boolean;
}

export function Model3D({
  url,
  position,
  rotation = [0, 0, 0],
  scale = 1,
  phaseOffset = 0,
  visible,
}: Model3DProps) {
  const groupRef = useRef<THREE.Group>(null);
  const { scene } = useGLTF(url, true, true); // useDraco=true, useMeshOpt=true
  const { gl } = useThree();

  // Clone scene to avoid mutating cached version
  const clonedScene = useMemo(() => scene.clone(true), [scene]);

  // Make all materials transparent (for fade out)
  useEffect(() => {
    clonedScene.traverse((child) => {
      if (child instanceof THREE.Mesh) {
        child.material.transparent = true;
        child.material.opacity = 1;
        // Cast shadow for realism
        child.castShadow = true;
        child.receiveShadow = true;
      }
    });
  }, [clonedScene]);

  // Pop-in spring animation (elastic bounce)
  const springs = useSpring({
    scale: visible ? scale : 0,
    opacity: visible ? 1 : 0,
    config: {
      tension: 280,
      friction: 18,
      mass: 1.2,
      // Elastic bounce
      clamp: false,
    },
    delay: phaseOffset * 200, // Stagger pop-in
  });

  // Scroll-driven animation (manual damp in useFrame)
  useFrame((_, delta) => {
    if (!groupRef.current) return;

    const offset = useScrollStore.getState().offset;
    // Hero section is roughly 0 to 0.15 of total page scroll
    const heroScrollProgress = Math.min(1, Math.max(0, (offset - phaseOffset * 0.05) / 0.15));

    // Phase 1 (0-0.4): scale down from 1 to 0.6
    const scalePhase = Math.min(1, heroScrollProgress / 0.4);
    const targetScale = THREE.MathUtils.lerp(scale, scale * 0.4, scalePhase);

    // Phase 2 (0.3-0.7): opacity fade from 1 to 0.5
    const opacityPhase = Math.min(1, Math.max(0, (heroScrollProgress - 0.3) / 0.4));
    const targetOpacity = THREE.MathUtils.lerp(1, 0.5, opacityPhase);

    // Phase 3 (0.6-1.0): lift up + rotate
    const liftPhase = Math.min(1, Math.max(0, (heroScrollProgress - 0.6) / 0.4));
    const targetY = THREE.MathUtils.lerp(position[1], position[1] + 1.5, liftPhase);
    const targetRotY = THREE.MathUtils.lerp(rotation[1], rotation[1] + Math.PI / 6, liftPhase);

    // Apply with damp for smooth, framerate-independent motion
    const dampFactor = 1 - Math.pow(0.001, delta); // ~0.25 damping at 60fps

    groupRef.current.scale.x = THREE.MathUtils.damp(
      groupRef.current.scale.x,
      targetScale * springs.scale.get(),
      8,
      delta
    );
    groupRef.current.scale.y = groupRef.current.scale.x;
    groupRef.current.scale.z = groupRef.current.scale.x;

    groupRef.current.position.y = THREE.MathUtils.damp(
      groupRef.current.position.y,
      targetY,
      8,
      delta
    );

    groupRef.current.rotation.y = THREE.MathUtils.damp(
      groupRef.current.rotation.y,
      targetRotY,
      6,
      delta
    );

    // Update material opacity (traverse once per frame)
    const finalOpacity = targetOpacity * springs.opacity.get();
    clonedScene.traverse((child) => {
      if (child instanceof THREE.Mesh) {
        child.material.opacity = finalOpacity;
      }
    });
  });

  // WebGL context loss handler (critical for iOS 17+)
  useEffect(() => {
    const canvas = gl.domElement;
    const handleContextLost = (e: Event) => {
      e.preventDefault();
      console.warn('[Model3D] WebGL context lost — will restore on focus');
    };
    const handleContextRestored = () => {
      console.log('[Model3D] WebGL context restored');
    };
    canvas.addEventListener('webglcontextlost', handleContextLost);
    canvas.addEventListener('webglcontextrestored', handleContextRestored);
    return () => {
      canvas.removeEventListener('webglcontextlost', handleContextLost);
      canvas.removeEventListener('webglcontextrestored', handleContextRestored);
    };
  }, [gl]);

  return (
    <a.group
      ref={groupRef}
      position={position}
      rotation={rotation}
      scale={springs.scale}
    >
      <primitive object={clonedScene} />
    </a.group>
  );
}

// Preload all 3 models at module load (starts download immediately)
useGLTF.preload('/models/The_chair_and_the_table_compressed.glb');
useGLTF.preload('/models/The_unfinished_sofa_compressed.glb');
useGLTF.preload('/models/the_dance_floor_and_the_light_holder_compressed.glb');
```

> ⚠️ **قواعد صارمة:**
> - **لا تستخدم `setState` في `useFrame`** — استخدم `useScrollStore.getState()` (مباشر بدون re-render)
> - **استخدم `THREE.MathUtils.damp`** (وليس `lerp`) للـ framerate independence
> - **`scene.clone(true)`** ضروري لتفادي تعديل الـ cached scene
> - **`phaseOffset`** يخلق staggered pop-in (LUT يظهر أولاً، ثم La Lounge، ثم Your Birthday)

'use client';

import { Canvas } from '@react-three/fiber';
import { AdaptiveDpr, AdaptiveEvents, Preload } from '@react-three/drei';
import { Suspense, useEffect, useRef, useState } from 'react';
import { Model3D } from './model-3d';
import { shouldEnable3D } from '@/lib/device-capabilities';
import { useScrollStore } from '@/stores/scroll-store';
import * as THREE from 'three';

interface HeroCanvasProps {
  /** Whether models have been "activated" (for pop-in) */
  modelsVisible: boolean;
}

export function HeroCanvas({ modelsVisible }: HeroCanvasProps) {
  const [mounted, setMounted] = useState(false);
  const [isVisible, setIsVisible] = useState(true);
  const containerRef = useRef<HTMLDivElement>(null);

  // Mount check (SSR safety)
  useEffect(() => {
    setMounted(true);
  }, []);

  // IntersectionObserver to pause rendering when out of viewport
  useEffect(() => {
    if (!containerRef.current) return;

    const observer = new IntersectionObserver(
      ([entry]) => {
        setIsVisible(entry.isIntersecting);
        useScrollStore.getState().setHeroVisible(entry.isIntersecting);
      },
      { threshold: 0, rootMargin: '100px' }
    );

    observer.observe(containerRef.current);
    return () => observer.disconnect();
  }, []);

  // Don't render on SSR or if device can't handle 3D
  if (!mounted || !shouldEnable3D()) return null;

  return (
    <div
      ref={containerRef}
      className="fixed inset-0 pointer-events-none z-20"
      aria-hidden="true"
    >
      {isVisible && (
        <Canvas
          camera={{
            position: [0, 0, 8],
            fov: 35,
            near: 0.1,
            far: 100,
          }}
          dpr={[1, 1.5]}  // ⚠️ Critical for mobile: clamp DPR
          frameloop="always"  // Required for Spring + damp animations
          gl={{
            antialias: false,  // Mobile: disable AA (use FXAA via postprocessing if needed)
            powerPreference: 'high-performance',
            alpha: true,  // Transparent background
            stencil: false,
            depth: true,
          }}
          onCreated={({ gl }) => {
            gl.toneMapping = THREE.ACESFilmicToneMapping;
            gl.toneMappingExposure = 1.0;
          }}
          // Performance helpers from drei
          style={{ background: 'transparent' }}
        >
          {/* Soft ambient light (preserves luxury dark theme) */}
          <ambientLight intensity={0.4} />

          {/* Subtle fill light from below (no harsh spotlights) */}
          <directionalLight
            position={[5, -5, 5]}
            intensity={0.15}
            color="#D4A574"  // Gold tint for luxury feel
          />

          <Suspense fallback={null}>
            {/* LUT — top-left corner */}
            <Model3D
              url="/models/The_chair_and_the_table_compressed.glb"
              position={[-3.5, 1.8, 0]}
              rotation={[0, -Math.PI / 8, 0]}
              scale={1.2}
              phaseOffset={0}
              visible={modelsVisible}
            />

            {/* La Lounge — top-right corner */}
            <Model3D
              url="/models/The_unfinished_sofa_compressed.glb"
              position={[3.5, 1.8, 0]}
              rotation={[0, Math.PI / 8, 0]}
              scale={1.0}
              phaseOffset={1}
              visible={modelsVisible}
            />

            {/* Your Birthday — bottom-right (visible on scroll) */}
            <Model3D
              url="/models/the_dance_floor_and_the_light_holder_compressed.glb"
              position={[2.5, -1.5, 0]}
              rotation={[0, Math.PI / 6, 0]}
              scale={0.9}
              phaseOffset={2}
              visible={modelsVisible}
            />
          </Suspense>

          {/* Performance optimizations */}
          <AdaptiveDpr pixelated={false} />
          <AdaptiveEvents />
          <Preload all />
        </Canvas>
      )}
    </div>
  );
}

'use client';

import { useEffect, useRef } from 'react';
import Lenis from 'lenis';
import { useScrollStore } from '@/stores/scroll-store';
import { shouldEnable3D } from '@/lib/device-capabilities';

export function LenisProvider({ children }: { children: React.ReactNode }) {
  const lenisRef = useRef<Lenis | null>(null);

  useEffect(() => {
    // Only enable Lenis if 3D is enabled (mobile-only smooth scroll)
    if (!shouldEnable3D()) return;

    const lenis = new Lenis({
      duration: 1.2,
      easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
      smoothWheel: true,
      // ⚠️ Critical: do NOT enable smoothTouch on iOS (causes jitter)
      smoothTouch: false,
      touchMultiplier: 1.5,
      infinite: false,
    });

    lenisRef.current = lenis;

    let rafId: number;
    const raf = (time: number) => {
      lenis.raf(time);
      rafId = requestAnimationFrame(raf);
    };
    rafId = requestAnimationFrame(raf);

    return () => {
      cancelAnimationFrame(rafId);
      lenis.destroy();
      lenisRef.current = null;
    };
  }, []);

  return <>{children}</>;
}

'use client';

import { useEffect, useState, lazy, Suspense } from 'react';
import { shouldEnable3D, isMobileDevice } from '@/lib/device-capabilities';
import { initScrollTracking } from '@/stores/scroll-store';

// Lazy load the 3D canvas (code splitting)
const HeroCanvas = lazy(() =>
  import('./hero-canvas').then((m) => ({ default: m.HeroCanvas }))
);

interface Hero3DSectionProps {
  children: React.ReactNode;  // The existing HTML hero content
}

export function Hero3DSection({ children }: Hero3DSectionProps) {
  const [enable3D, setEnable3D] = useState(false);
  const [modelsVisible, setModelsVisible] = useState(false);

  // Initialize scroll tracking + 3D decision
  useEffect(() => {
    const cleanup = initScrollTracking();
    setEnable3D(shouldEnable3D() && isMobileDevice());

    // Trigger pop-in after a short delay (let models preload)
    const timer = setTimeout(() => setModelsVisible(true), 300);

    return () => {
      cleanup();
      clearTimeout(timer);
    };
  }, []);

  return (
    <section className="relative">
      {/* 3D Canvas overlay (fixed, pointer-events-none) */}
      {enable3D && (
        <Suspense fallback={null}>
          <HeroCanvas modelsVisible={modelsVisible} />
        </Suspense>
      )}

      {/* Original HTML hero content (always present, for SEO + fallback) */}
      {children}
    </section>
  );
}

'use client';

import { useEffect } from 'react';
import { useGLTF } from '@react-three/drei';

const MODEL_URLS = [
  '/models/The_chair_and_the_table_compressed.glb',
  '/models/The_unfinished_sofa_compressed.glb',
  '/models/the_dance_floor_and_the_light_holder_compressed.glb',
];

/**
 * Cleans up useGLTF cache when navigating away from the homepage.
 * Mount this ONCE on the homepage (prevents memory leaks).
 */
export function HeroCleanup() {
  useEffect(() => {
    return () => {
      // Clear cache on unmount (when navigating away from homepage)
      MODEL_URLS.forEach((url) => {
        try {
          useGLTF.clear(url);
        } catch (e) {
          console.warn('[HeroCleanup] Failed to clear cache for', url, e);
        }
      });
    };
  }, []);

  return null;
}

'use client';

import { Perf } from 'r3f-perf';
import { useState, useEffect } from 'react';

export function PerfMonitor() {
  const [show, setShow] = useState(false);

  useEffect(() => {
    // Toggle with keyboard shortcut (Ctrl+Shift+P) in development
    if (process.env.NODE_ENV !== 'development') return;

    const handler = (e: KeyboardEvent) => {
      if (e.ctrlKey && e.shiftKey && e.key === 'P') {
        e.preventDefault();
        setShow((s) => !s);
      }
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, []);

  if (!show) return null;

  return (
    <Perf
      position="top-left"
      minimal={false}
      showGraph={true}
      colorBlind={false}
    />
  );
}
```
````
